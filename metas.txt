Options names:

-d	--directory	 = diretório
-fn	--filename  	 = nome do arquivo
-dt	--directoryto 	 = diretório para

Commands:

move = move um arquivo pra outro diretório
	-- possibilitar mover mais de um

delete	= deleta um arquivo
	-- possibilitar deletar mais de um
	-- opções para deletar por data
	-- opções para deletar por tamanho

listFiles = lista os arquivos de um diretório
	-- lista nome
	-- lista o tamanho
	-- lista data de mod
		-img	--image = lista somente as imagens
		-doc	--documents = lista somente doc/docx, sheets e slides
		-zp	--zips = lista todos os zips e rar
		-c	--codes = lista todos os códigos

organize = organiza os arquivos de um diretório
